// Firebase Storage Security Rules
// Ga naar Firebase Console → Storage → Rules en zorg dat je dit hebt:

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Sta toe dat geauthenticeerde gebruikers hun eigen foto's uploaden
    match /users/{userId}/photos/{allPaths=**} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Voor alle andere bestanden - alleen lezen toegestaan voor geauthenticeerde gebruikers
    match /{allPaths=**} {
      allow read: if request.auth != null;
    }
  }
}
